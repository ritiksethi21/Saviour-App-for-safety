1    package com.example.driversafety;

3    import android.os.AsyncTask;
4    
5    import com.google.android.gms.maps.CameraUpdateFactory;
6    import com.google.android.gms.maps.GoogleMap;
7    import com.google.android.gms.maps.model.BitmapDescriptorFactory;
8    import com.google.android.gms.maps.model.LatLng;
9    import com.google.android.gms.maps.model.MarkerOptions;
10   
11   import java.io.IOException;
12   import java.util.HashMap;
13   import java.util.List;
14   
15   public class GetNearByPlaces extends AsyncTask<Object,String,String> {
16       private String googleplaceData, url;
17       private GoogleMap mMap;
18   
19       @Override
20       protected String doInBackground(Object... objects) {
21           mMap=(GoogleMap) objects[0];
22           url=(String) objects[1];
23           DownloadUrl downloadUrl=new DownloadUrl();
24           try {
25               googleplaceData=downloadUrl.ReadTheURL(url);
26           } catch (IOException e) {
27               e.printStackTrace();
28           }
29           return googleplaceData;
30       }
31   
32       @Override
33       protected void onPostExecute(String s) {
34           List<HashMap<String,String>> nearByPlacesList=null;
35           DataParser dataParser=new DataParser();
36           nearByPlacesList=dataParser.parse(s);
37           DisplayNearbyPlaces(nearByPlacesList);
38   
39       }
40       private void DisplayNearbyPlaces(List<HashMap<String,String>> nearByPlacesList)
41       {
42           for (int i=0;i<nearByPlacesList.size();i++)
43           {
44               MarkerOptions markerOptions=new MarkerOptions();
45               HashMap<String,String>googleNearbyPlace=nearByPlacesList.get(i);
46               String nameOfPlace=googleNearbyPlace.get("place_name");
47               String vicinity=googleNearbyPlace.get("vicinity");
48               double lat=Double.parseDouble(googleNearbyPlace.get("lat"));
49               double lng=Double.parseDouble(googleNearbyPlace.get("lng"));
50               LatLng latLng=new LatLng(lat,lng);
51               markerOptions.position(latLng);
52               markerOptions.title(nameOfPlace+":"+vicinity);
53               markerOptions.icon(BitmapDescriptorFactory.defaultMarker(BitmapDescriptorFactory.HUE_GREEN));
54               mMap.addMarker(markerOptions);
55               mMap.moveCamera(CameraUpdateFactory.newLatLng(latLng));
56               mMap.animateCamera(CameraUpdateFactory.zoomBy(10));
57           }
58       }
59   }

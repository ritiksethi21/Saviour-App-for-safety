47       private List<HashMap<String,String>> getAllNearbyPlaces(JSONArray jsonArray)
48       {
49           int counter=jsonArray.length();
50           List<HashMap<String,String>>NearbyPlaceList=new ArrayList<>();
51           HashMap<String,String>NearbyPlaceMap=null;
52   
53           for(int i=0;i<counter;i++)

160      @Override
161      public void onLocationChanged(@NonNull Location location) {
162  
163          latitude=location.getLatitude();
164          longitude=location.getLongitude();
165           lastLocation=location;
166           if(currentUserLocationMarker!=null)
167           {
168               currentUserLocationMarker.remove();
169           }
170           LatLng latLng=new LatLng(location.getLatitude(),location.getLongitude());
171           MarkerOptions markerOptions=new MarkerOptions();
172           markerOptions.position(latLng);
173           markerOptions.title("Your Current Location");
174           markerOptions.icon(BitmapDescriptorFactory.defaultMarker(BitmapDescriptorFactory.HUE_VIOLET));
175           currentUserLocationMarker=mMap.addMarker(markerOptions);
176           mMap.moveCamera(CameraUpdateFactory.newLatLng(latLng));
177           mMap.animateCamera(CameraUpdateFactory.zoomBy(12));
178           if(googleApiClient!=null)
179           {
180               LocationServices.FusedLocationApi.removeLocationUpdates(googleApiClient,this);
181           }
182      }
